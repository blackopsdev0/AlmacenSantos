
CREATE TABLE DET_VENTAS (
  ID int(11) NOT NULL AUTO_INCREMENT,
  ID_VENTA int(11) NOT NULL,
  ID_PRODUCTO int(11) NOT NULL,
  ID_MARCA int(11) NOT NULL,
  CANTIDAD int(11) NOT NULL,
  PREC_UNITARIO decimal(10,2) NOT NULL,
  PREC_TOTAL decimal(10,2) NOT NULL,
  STATUS_FLAG int(2) NOT NULL,
  START_DATE timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  LAST_UPDATE_USER varchar(50) DEFAULT NULL,
  PRIMARY KEY (ID)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
ALTER TABLE DET_VENTAS AUTO_INCREMENT=100;
alter table DET_VENTAS add foreign key(ID_VENTA) references VENTAS(ID);
alter table DET_VENTAS add foreign key(ID_PRODUCTO) references CAT_PRODUCTOS(ID);
alter table DET_VENTAS add foreign key(ID_MARCA) references CAT_MARCAS(ID);
